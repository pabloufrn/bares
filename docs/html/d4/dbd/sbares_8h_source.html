139 void infix_to_postfix( sc::vector< Token > & infix, sc::vector<Token> & postfix )
140 {
141  pl::stack< Token > s; // Estrutura de dado auxiliar.
142 
143  // Processa cada symbolo.
144  for( const auto & symbol : infix )
145  {
146  // operando entra de qualquer jeito
147  if ( is_operand( symbol.type ) )
148  {
149  postfix.push_back( symbol ); // envia o simbolo para a fila de simbolos de saída.
150  }
151  else if ( is_opening_scope( symbol.type ) )
152  {
153  s.push( symbol ); // sempre coloca na "lista de espera".
154  }
155  else if ( is_closing_scope( symbol.type ) )
156  {
157  // Remove todas as operações pendentes.
158  while( not is_opening_scope( s.top().type ) )
159  {
160  // Remove o operador e envia para a expresão postfix.
161  postfix.push_back( s.top() );
162  s.pop();
163  }
164  // Remove o parentese aberto de dentro da pilha.
165  s.pop();
166  }
167  else if ( is_operator( symbol.type ) )
168  {
169  // Sai os operadores que estão esperando com precedencia maior ou igual.
170  while ( not s.empty() and has_higher_or_eq_precedence( s.top().value, symbol.value ) )
171  {
172  postfix.push_back(s.top()); // envia para a saída.
173  s.pop(); // remove o operaodr.
174  }
175  // O simbolo sempre vai para a sala de espera caso seja menor.
176  s.push( symbol ) ;
177  }
178  }
179 
180  // Remove qualquer operador que esteja na fila.
181  while ( not s.empty() )
182  {
183  postfix.push_back(s.top());
184  s.pop();
185  }
